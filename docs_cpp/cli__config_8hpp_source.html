65 pointcloudcrafter::tools::formats::FileFormat get_load_format() const
66 {

93 void add_modifier_options(CLI::App * app)
94 {
95 // Output
96 app->add_flag("--timestamps", timestamps, "Save point cloud timestamps to a text file")
97 ->group("Output");
98
99 app->add_flag("--sequential-name", sequential_names, "Use sequential file names")
100 ->group("Output");
101
102 // Frame selection
103 app->add_option("-m,--max-frames", max_frames, "Maximum number of frames (-1 = unlimited)")
104 ->group("General");
105
106 app->add_option("-j,--skip-frames", skip_frames, "Number of frames to skip at the beginning")
107 ->group("General");
108
109 app->add_option("-s,--stride-frames", stride_frames, "Write every Nth frame")
110 ->group("General");
111
112 // File format
113 app->add_flag("--save-pcd", save_pcd, "Save PCD files (default)")
114 ->group("File Format");
115 app->add_flag("--save-ply", save_ply, "Save PLY files")
116 ->group("File Format");
117 app->add_flag("--save-txt", save_txt, "Save TXT ASCII files")
118 ->group("File Format");
119 app->add_flag("--save-kitti", save_kitti, "Save KITTI binary files")
120 ->group("File Format");
121 app->add_flag("--save-nuscenes", save_nuscenes, "Save nuScenes binary files")
122 ->group("File Format");
123
124 // Transforms
125 app->add_option(
126 "--tf, --transform-file", transform_file,
127 "TXT file with transform ([frame_id] r1 r2 r3 x r4 r5 r6 y r7 r8 r9 z)")
128 ->group("Transforms");
129
130 // Filtering
131 app->add_option("--cb, --crop-box", cropbox, "Crop box [xmin ymin zmin xmax ymax zmax]")
132 ->expected(6)
133 ->type_name("FLOAT FLOAT FLOAT FLOAT FLOAT FLOAT")
134 ->group("Filtering");
135
136 app->add_option("--cs, --crop-sphere", cropsphere, "Crop to sphere with given radius")
137 ->type_name("FLOAT")
138 ->group("Filtering");
139
140 app->add_option("--cc, --crop-cylinder", cropcylinder, "Crop to cylinder with given radius")
141 ->type_name("FLOAT")
142 ->group("Filtering");
143
144 app->add_flag("--inverse-crop", inverse_crop, "Inverse crop filters")
145 ->group("Filtering");
146
147 app->add_option("--vf, --voxel-filter", voxelfilter, "Voxel size [x y z]")
148 ->expected(3)
149 ->type_name("FLOAT FLOAT FLOAT")
150 ->group("Filtering");
151
152 app->add_option(
153 "--orf, --outlier-radius-filter", outlier_radius_filter,
154 "Radius outlier removal [radius min_neighbors]")
155 ->expected(1)
156 ->type_name("FLOAT INT")
157 ->group("Filtering");
158
159 app->add_option(
160 "--osf, --outlier-stat-filter", outlier_stat_filter,
161 "Statistical outlier removal [threshold mean_k]")
162 ->expected(1)
163 ->type_name("FLOAT INT")
164 ->group("Filtering");
165 }

171struct FileConfig : public ModifierConfig
172{
173 std::string input_path{};
174 std::vector<double> translation{};
175 std::vector<double> rotation{};
176 bool degree{false};
177 bool merge_clouds{false};
178
183 void add_cli_options(CLI::App * app)
184 {
185 app->add_option("input-path", input_path, "Path to point cloud file or directory")
186 ->required()
187 ->group("Required");
188
189 app->add_option("out-dir", out_dir, "Output directory for point cloud file(s)")
190 ->required()
191 ->group("Required");
192
193 app->add_option("urdf-file", urdf_file, "URDF file with transformations");
194
195 app->add_flag("--load-pcd", load_pcd, "Load PCD files (default)")
196 ->group("File Format");
197 app->add_flag("--load-ply", load_ply, "Load PLY files")
198 ->group("File Format");
199 app->add_flag("--load-txt", load_txt, "Load TXT ASCII files")
200 ->group("File Format");
201 app->add_flag("--load-kitti", load_kitti, "Load KITTI binary files")
202 ->group("File Format");
203 app->add_flag("--load-nuscenes", load_nuscenes, "Load nuScenes binary files")
204 ->group("File Format");
205 app->add_flag("--load-obj", load_obj, "Load OBJ files")
206 ->group("File Format");
207
208 app->add_option("-t,--translation", translation, "Translation [x y z]")
209 ->expected(3)
210 ->type_name("FLOAT FLOAT FLOAT")
211 ->group("Transforms");
212
213 app->add_option("-r,--rotation", rotation, "Rotation [roll pitch yaw]")
214 ->expected(3)
215 ->type_name("FLOAT FLOAT FLOAT")
216 ->group("Transforms");
217
218 app->add_flag("--deg", degree, "Rotation in degrees instead of radians")
219 ->group("Transforms");
220
221 app->add_flag("--merge-clouds", merge_clouds, "Merge all point clouds into a single file")
222 ->group("Transforms");
223
224 add_modifier_options(app);
225 }
