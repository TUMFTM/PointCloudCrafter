97 explicit RosbagReader(const std::string & BAG_PATH) { reader_.open(BAG_PATH); }
104 template <typename T>
105 void add_listener(
106 const std::string & topic_name, std::function<void(const RosbagReaderMsg<T> &)> callback)
107 {
108 handlers_[topic_name].push_back(std::make_unique<SerializedMessageHandler<T>>(callback));
109 }
114 void set_state(bool state) { running_ = state; }
119 void process()
120 {
121 int64_t first_msg_time = -1;
122 rclcpp::Clock wall_clock{};
123
124 // Configure the reader to only read the topics we are interested in
125 std::vector<std::string> topics{};
126 topics.reserve(handlers_.size());
127 for (auto & entry : handlers_) {
128 topics.push_back(entry.first);
129 }
130 reader_.set_filter({topics});
131
132 // Initialize progress logging
133 double total_duration = static_cast<double>((reader_.get_metadata().duration.count()) / 1.0e9);
134 running_ = true;
135
136 // Process the bag files
137 while (reader_.has_next() && rclcpp::ok() && running_) {
138 auto bag_msg = reader_.read_next();
139 rclcpp::SerializedMessage ser_msg{*bag_msg->serialized_data};
140
141 // Get the first message time
142 if (first_msg_time < 0) {
143 first_msg_time = bag_msg->time_stamp;
144 }
145
146 // Log progress
147 double time_done = static_cast<double>((bag_msg->time_stamp - first_msg_time) / 1e9);
148 RCLCPP_INFO_THROTTLE(
149 logger_, wall_clock, 1000, "Processed %.3f sec of bag [% 5.1f%%]", time_done,
150 100 * time_done / total_duration);
151
152 // Call the handlers
153 auto it = handlers_.find(bag_msg->topic_name);
154 if (it == handlers_.end()) {
155 continue;
156 }
157
158 for (auto & handler : it->second) {
159 handler->handle_serialized_message(*bag_msg, ser_msg);
160 }
161 }
162 }
163
164private:
165 // Map to store the handlers for each topic
166 std::unordered_map<std::string, std::vector<std::shared_ptr<SerializedMessageHandlerBase>>>
167 handlers_;
168 // Logger to print progress
169 rclcpp::Logger logger_ = rclcpp::get_logger("rosbag_reader");
170 // Reader to read the bag file
171 rosbag2_cpp::Reader reader_;
172 // State flag
173 bool running_{true};
174};
